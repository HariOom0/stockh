   161→// ─── Sector rotation rank (lower = stronger) ─────────────────────────
   162→function trendRank(trend: string): number {
   163→  const ranks: Record<string, number> = {
   164→    "Bullish": 0,
   165→    "Rotating In": 1,
   166→    "Neutral": 2,
   167→    "Bearish": 3,
   168→    "Rotating Out": 4,
   169→  };
   170→  return ranks[trend] ?? 2;
   171→}

   173→// ─── Match a stock's sector to the nearest sector insight ─────────────
   174→function matchSectorInsight(
   175→  stockSector: string | undefined,
   176→  insights: SectorInsight[]
   177→): SectorInsight | null {
   178→  if (!stockSector || insights.length === 0) return null;
   179→  // Try exact match first, then partial match on first keyword
   180→  const lower = stockSector.toLowerCase();
   181→  const exact = insights.find(
   182→    (s) => s.sector.toLowerCase() === lower
   183→  );
   184→  if (exact) return exact;
   185→  const partial = insights.find((s) => {
   186→    const keywords = s.sector.toLowerCase().split(/[\s&]+/).filter((w) => w.length > 2);
   187→    return keywords.some((kw) => lower.includes(kw));
   188→  });
   189→  return partial || null;
   190→}
