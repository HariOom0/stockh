   162→// ─── Sector rotation rank (lower = stronger) ─────────────────────────
   163→function trendRank(trend: string): number {
   164→  const ranks: Record<string, number> = {
   165→    "Bullish": 0,
   166→    "Rotating In": 1,
   167→    "Neutral": 2,
   168→    "Bearish": 3,
   169→    "Rotating Out": 4,
   170→  };
   171→  return ranks[trend] ?? 2;
   172→}
   173→
   174→// ─── Match a stock's sector to the nearest sector insight ─────────────
   175→function matchSectorInsight(
   176→  stockSector: string | undefined,
   177→  insights: SectorInsight[]
   178→): SectorInsight | null {
   179→  if (!stockSector || insights.length === 0) return null;
   180→  // Try exact match first, then partial match on first keyword
   181→  const lower = stockSector.toLowerCase();
   182→  const exact = insights.find(
   183→    (s) => s.sector.toLowerCase() === lower
   184→  );
   185→  if (exact) return exact;
   186→  const partial = insights.find((s) => {
   187→    const keywords = s.sector.toLowerCase().split(/[\s&]+/).filter((w) => w.length > 2);
   188→    return keywords.some((kw) => lower.includes(kw));
   189→  });
   190→  return partial || null;
   191→}

   222→              {h}

   225→        </tr>
